## Technical Architecture

### Core Classes
```typescript
// Abstract base class
export abstract class Biome {
  constructor(config: BiomeConfig);
  abstract generateLayers(width: number, height: number): string[];
  getSVG(width: number, height: number): string;
}

// Singleton factory
export class BiomeFactory {
  private biomeRegistry: Map<BiomeType, () => Biome>;
  createBiome(biomeType: BiomeType): Biome;
}

// Detection engine
export class BiomeDetector {
  static detectBiome(
    location: Location,
    state?: string,
    elevation?: number
  ): Promise<BiomeDetectionResult>;
}

// Main system
export class BiomeSystem {
  static async detectAndRender(
    location: Location,
    state?: string
  ): Promise<string>;
}
```

## Usage Examples

### Basic Usage
```typescript
import { BiomeSystem } from './biomes';

const location = { lat: 38.9072, lng: -77.0369 }; // Washington DC
const state = 'DC';

const biomeSVG = await BiomeSystem.detectAndRender(location, state);
// Returns: SVG string for Appalachian mountain biome
```

### React Integration
```tsx
import { BiomeRenderer } from './biomes';

function App() {
  return (
    <BiomeRenderer
      location={{ lat: 37.7749, lng: -122.4194 }}
      state="CA"
      width={800}
      height={600}
    />
  );
}
```

### Custom Detection
```typescript
import { BiomeDetector, BiomeFactory } from './biomes';

const result = await BiomeDetector.detectBiome(
  { lat: 36.1699, lng: -115.1398 }, // Las Vegas
  'NV',
  600 // elevation in meters
);

console.log(result);
// {
//   biomeType: 'desert_mojave',
//   confidence: 0.85,
//   metadata: { state: 'NV', elevation: 600 }
// }

const factory = BiomeFactory.getInstance();
const biome = factory.createBiome(result.biomeType);
const svg = biome.getSVG(800, 600);
```
