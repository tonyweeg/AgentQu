Zipquest Platform - Complete Technical Specification
Executive Summary
Zipquest is a real-time activity discovery platform that helps users find last-minute things to do based on their location, accessibility needs, and current context (weather, time of day). The platform prioritizes accessibility-first design and deterministic, explainable results over black-box algorithms.
Core Value Proposition
"What should I do right now?" - answered in under 2 seconds with personalized, accessible suggestions for venues, events, hikes, and daytime activities.
Platform Principles

Accessibility First - Every activity must have complete accessibility metadata
Deterministic Scoring - Same inputs always produce same outputs, fully explainable
Real-Time Context - Weather, time, and location-aware suggestions
Privacy-Focused - Minimal data collection, user control over location
Free Tier Viable - Architecture stays within Firebase/GCP free tiers during development


System Architecture Overview

exports.discoverActivities = functions.https.onCall(async (data, context) => {
  const startTime = Date.now();
  
  // === VALIDATION ===
  const { lat, lng, radius = 10, filters = {} } = data;

  }
  
  // === USER CONTEXT ===
  const userId = context.auth?.uid;
  const user = userId

  // === CURRENT CONTEXT ===
  const currentContext = {
    userLat: lat,
    userLng: lng,
    timestamp: Date.now(),
    hour: new Date().getHours(),
    dayOfWeek: new Date().getDay(),
    isWeekend: [0, 6].includes(new Date().getDay()),
    weather: await fetchWeather(lat, lng)
  };
  
  // === GEOSPATIAL QUERY ===
  const bounds = calculateGeohashBounds(lat, lng, radius);
  
  let query = db.collection('activities')
    .where('geohash', '>=', bounds.lower)
    .where('geohash', '<=', bounds.upper);
  
  // Apply open now filter by default
  if (!filters.includeClosedVenues) {
    query = query.where('hoursToday.openNow', '==', true);
  }
  
  const snapshot = await query.get();
  
  let activities = snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));
  
  console.log(`Found ${activities.length} activities in radius`);
  
  // === HARD ACCESSIBILITY FILTERS ===
  activities = applyAccessibilityFilters(activities, user.accessibilityNeeds);
  console.log(`${activities.length} activities after accessibility filters`);
  
  // === CATEGORY FILTERS ===
  if (filters.categories && filters.categories.length > 0) {
    activities = activities.filter(a => 
      filters.categories.some(cat => a.categories.includes(cat))
    );
  }
  
  if (user.preferences.avoidCategories.length > 0) {
    activities = activities.filter(a =>
      !user.preferences.avoidCategories.some(cat => a.categories.includes(cat))
    );
  }
  
  // === CALCULATE DISTANCE ===
  activities = activities.map(activity => ({
    ...activity,
    distance: calculateDistance(lat, lng, activity.lat, activity.lng)
  }));
  
  // === DISTANCE FILTER ===
  activities = activities.filter(a =>

  console.log(`${activities.length} activities within distance`);
  
  // === SCORE EACH ACTIVITY ===

    return {
      ...activity,
      totalScore: scoring.totalScore,
      scoreBreakdown: scoring.breakdown,

    if (b.totalScore !== a.totalScore) {

    }
    // Tiebreaker 1: Distance
    if (a.distance !== b.distance) {
      return a.distance - b.distance;
    }
    // Tiebreaker 2: Review count
    return b.reviewCount - a.reviewCount;

  // === DIVERSITY FILTER ===

  const results = diverse.slice(0, 20).map(activity => ({

/**
 * Helper: Apply hard accessibility filters
 * Returns only activities that meet REQUIRED accessibility needs
 */
function applyAccessibilityFilters(activities, needs) {
  if (!needs || Object.keys(needs).length === 0) {
    return activities;
  }
  
  return activities.filter(activity => {
    const access = activity.accessibility;
    
    // Wheelchair requirement (HARD FILTER)
    if (needs.requiresWheelchair) {
      if (!access.wheelchairAccessible) return false;
      if (!access.accessibleRestrooms) return false;
    }
    
    // Service animal requirement (HARD FILTER)
    if (needs.requiresServiceAnimal) {
      if (!access.serviceAnimalFriendly) return false;
    }
    
    // Mobility level requirement (HARD FILTER)
    if (needs.mobilityLevel) {
      const mobilityLevels = { easy: 1, moderate: 2, difficult: 3 };
      const activityLevel = mobilityLevels[access.mobilityLevel];
      const userMaxLevel = mobilityLevels[needs.mobilityLevel];
      
      if (activityLevel > userMaxLevel) return false;
    }
    
    // Sensory requirements (HARD FILTER)
    if (needs.requiresSensoryFriendly) {
      if (!access.sensoryFriendly.quietSpace) return false;
    }
    
    return true;
  });
}

/**
 * Helper: Calculate deterministic score
 * CRITICAL: This function must be pure - same inputs = same outputs
 */
function calculateScore(activity, user, context) {
  const scores = {
    accessibility: 0,
    distance: 0,
    time: 0,
    cost: 0,
    quality: 0,
    weather: 0
  };
  
  const weights = {
    accessibility: 0.30,
    distance: 0.25,
    time: 0.15,
    cost: 0.15,
    quality: 0.10,
    weather: 0.05
  };
  
  // 1. ACCESSIBILITY SCORE (0-100)
  scores.accessibility = calculateAccessibilityScore(
    activity.accessibility,
    user.accessibilityNeeds
  );
  
  // 2. DISTANCE SCORE (0-100)
  if (activity.distance <= 1) scores.distance = 100;
  else if (activity.distance <= 3) scores.distance = 80;
  else if (activity.distance <= 5) scores.distance = 60;
  else if (activity.distance <= 10) scores.distance = 40;
  else if (activity.distance <= 20) scores.distance = 20;
  else scores.distance = 0;
  
  // 3. TIME SCORE (0-100)
  if (activity.hoursToday.openNow) {
    const hoursOpen = calculateHoursUntilClose(activity.hoursToday.close, context.hour);
    const needed = (activity.details?.estimatedTime || 60) / 60;
    
    if (hoursOpen >= needed * 2) scores.time = 100;
    else if (hoursOpen >= needed) scores.time = 80;
    else scores.time = 40;
  } else {
    scores.time = 0;
  }
  
  // 4. COST SCORE (0-100)
  if (activity.cost.free) {
    scores.cost = 100;
  } else if (user.preferences.pricePreference.maxBudget) {
    const maxBudget = user.preferences.pricePreference.maxBudget;
    const price = activity.cost.price || activity.cost.priceRange?.min || 0;
    
    if (price <= maxBudget * 0.5) scores.cost = 100;
    else if (price <= maxBudget) scores.cost = 80;
    else if (price <= maxBudget * 1.2) scores.cost = 50;
    else scores.cost = 20;
  } else {
    scores.cost = 70; // neutral
  }
  
  // 5. QUALITY SCORE (0-100)
  if (activity.reviewCount >= 100) {
    scores.quality = (activity.rating / 5.0) * 100;
  } else if (activity.reviewCount >= 10) {
    scores.quality = ((activity.rating / 5.0) * 100) * 0.8;
  } else {
    scores.quality = 50; // neutral for unreviewed
  }
  
  // 6. WEATHER SCORE (0-100)
  scores.weather = calculateWeatherScore(activity, context.weather);
  
  // Calculate weighted total
  const totalScore = Object.keys(weights).reduce((sum, key) => {
    return sum + (scores[key] * weights[key]);
  }, 0);
  
  return {
    totalScore: Math.round(totalScore),
    breakdown: scores,
    weights: weights
  };
}

/**
 * Helper: Calculate accessibility score with soft preferences
 */
function calculateAccessibilityScore(activityAccess, userNeeds) {
  if (!userNeeds || Object.keys(userNeeds).length === 0) {
    return 100; // No needs = all activities accessible
  }
  
  let score = 100;
  
  // Note: Hard requirements already filtered in applyAccessibilityFilters
  // This function handles SOFT PREFERENCES
  
  // Visual preferences
  if (userNeeds.preferVisualAccommodations) {
    if (activityAccess.visualImpairment.suitable) {
      score += 0; // Already perfect
    } else {
      score -= 20; // Less ideal but not filtered
    }
  }
  
  // Hearing preferences
  if (userNeeds.preferHearingAccommodations) {
    if (activityAccess.hearingImpairment.suitable) {
      score += 0;
    } else {
      score -= 20;
    }
  }
  
  // Cognitive preferences
  if (userNeeds.preferCognitiveSupport) {
    if (activityAccess.cognitiveAccessibility.suitable) {
      score += 0;
    } else {
      score -= 15;
    }
  }
  
  return Math.max(0, Math.min(100, score));
}

/**
 * Helper: Calculate weather score
 */
function calculateWeatherScore(activity, weather) {
  // Indoor activities not affected by weather
  if (!activity.weatherRequirements.requiresGoodWeather) {
    return 100;
  }
  
  const { condition, temp } = weather;
  const reqs = activity.weatherRequirements;
  
  // Temperature check
  if (temp < reqs.minTemp || temp > reqs.maxTemp) {
    return 30; // Poor weather match
  }
  
  // Condition check
  if (reqs.idealConditions.includes(condition)) {
    return 100; // Perfect weather
  }
  
  if (reqs.avoidRain && condition.includes('rain')) {
    return 20; // Bad weather
  }
  
  if (condition === 'cloudy') {
    return 70; // Acceptable
  }
  
  return 60; // Default
}

/**
 * Helper: Diversify results by type and category
 */
function diversifyResults(scored, options) {
  const { maxPerType, maxPerPrimaryCategory } = options;
  const typeCounts = {};
  const categoryCounts = {};
  const results = [];
  
  for (const activity of scored) {
    // Check type limit
    typeCounts[activity.type] = (typeCounts[activity.type] || 0);
    if (typeCounts[activity.type] >= maxPerType) continue;
    
    // Check category limit
    const cat = activity.primaryCategory;
    categoryCounts[cat] = (categoryCounts[cat] || 0);
    if (categoryCounts[cat] >= maxPerPrimaryCategory) continue;
    
    results.push(activity);
    typeCounts[activity.type]++;
    categoryCounts[cat]++;
    
    if (results.length >= 40) break; // Enough candidates
  }
  
  return results;
}

/**
 * Rate limiting using Firestore transactions
 */
async function checkAndIncrementRateLimit(apiName, maxPerSecond) {
  const limitsRef = db.collection('cache_control').doc(apiName);
  
  return db.runTransaction(async (transaction) => {
    const doc = await transaction.get(limitsRef);
    const now = Date.now();
    const windowSize = 1000; // 1 second
    
    let data = doc.exists ? doc.data() : {
      rateLimits: {
        perSecond: maxPerSecond,
        currentWindow: {
          windowStart: now,
          requestCount: 0
        }
      }
    };
    
    const window = data.rateLimits.currentWindow;
    
    // Reset if new window
    if (now - window.windowStart > windowSize) {
      window.windowStart = now;
      window.requestCount = 0;
    }
    
    // Check limit
    if (window.requestCount >= maxPerSecond) {
      return false; // Rate limited
    }
    
    // Increment
    window.requestCount++;
    transaction.set(limitsRef, data);
    
    return true; // Allowed
  });
}

 * Update user affinities based on interactions

exports.updateUserAffinities = functions.firestore
  .document('interactions/{interactionId}')
  .onCreate(async (snap, context) => {
    
    const interaction = snap.data();
    const userId = interaction.userId;
    
    if (!userId) return; // Anonymous user
    
    const userRef = db.collection('users').doc(userId);
    
    return db.runTransaction(async (transaction) => {
      const userDoc = await transaction.get(userRef);
      
      if (!userDoc.exists) return;
      
      const userData = userDoc.data();
      const affinities = userData.affinities || {};
      
      // Learning rates based on action type
      const learningRates = {
        view: 0.05,
        click: 0.15,
        book: 0.30,
        favorite: 0.25,
        dismiss: -0.10
      };
      
      const learningRate = learningRates[interaction.action] || 0.05;
      const targetValue = interaction.action === 'dismiss' ? 0.3 : 1.0;
      
      // Update affinity for each category
      interaction.activitySnapshot.categories.forEach(category => {
        const currentAffinity = affinities[category] || 0.5;
        
        // Exponential moving average
        affinities[category] = 
          currentAffinity * (1 - Math.abs(learningRate)) +
          targetValue * learningRate;
        
        // Clamp to [0, 1]
        affinities[category] = Math.max(0, Math.min(1, affinities[category]));
      });
      
      transaction.update(userRef, {
        affinities: affinities,
        'stats.totalViews': admin.firestore.FieldValue.increment(
          interaction.action === 'view' ? 1 : 0
        ),
        'stats.totalClicks': admin.firestore.FieldValue.increment(
          interaction.action === 'click' ? 1 : 0
        ),
        'stats.totalBookings': admin.firestore.FieldValue.increment(
          interaction.action === 'book' ? 1 : 0
        ),
        lastActive: admin.firestore.FieldValue.serverTimestamp()
      });
    });
  });

javascript/**
 * Calculate distance between two points (Haversine formula)
 */
function calculateDistance(lat1, lng1, lat2, lng2) {
  const R = 3959; // Earth radius in miles
  const dLat = toRad(lat2 - lat1);
  const dLng = toRad(lng2 - lng1);
  
  const a = 
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) *
    Math.sin(dLng / 2) * Math.sin(dLng / 2);
  
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c;
}

function toRad(degrees) {
  return degrees * (Math.PI / 180);
}

/**
 * Get hot zones (areas with recent user activity)
 */
async function getHotZones() {
  const oneHourAgo = Date.now() - (60 * 60 * 1000);
  
  // Query recent user locations
  const usersSnapshot = await db.collection('users')
    .where('location.lastUpdated', '>', oneHourAgo)
    .limit(100)
    .get();
  
  // Group by geohash4 to find hot zones
  const zoneMap = new Map();
  
  usersSnapshot.docs.forEach(doc => {
    const userData = doc.data();
    const geohash4 = userData.homeGeohash?.slice(0, 4);
    
    if (geohash4) {
      const zone = zoneMap.get(geohash4) || {
        geohash: geohash4,
        lat: userData.location.lat,
        lng: userData.location.lng,
        count: 0
      };
      zone.count++;
      zoneMap.set(geohash4, zone);
    }
  });
  
  // Return top 20 hot zones
  return Array.from(zoneMap.values())
    .sort((a, b) => b.count - a.count)
    .slice(0, 20);
}

Location Hook
typescript// src/hooks/useLocation.ts
import { useState, useCallback } from 'react';

interface Location {
  lat: number;
  lng: number;
  accuracy: number;
}

export function useLocation() {
  const [location, setLocation] = useState<Location | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  
  const requestLocation = useCallback(() => {
    if (!navigator.geolocation) {
      setError(new Error('Geolocation not supported'));
      return;
    }
    
    setLoading(true);
    setError(null);
    
    navigator.geolocation.getCurrentPosition(
      (position) => {
        setLocation({
          lat: position.coords.latitude,
          lng: position.coords.longitude,
          accuracy: position.coords.accuracy
        });
        setLoading(false);
      },
      (err) => {
        setError(err as GeolocationPositionError);
        setLoading(false);
      },
      {
        enableHighAccuracy: false, // Faster, uses WiFi/IP
        timeout: 5000,
        maximumAge: 60000 // Cache for 1 minute
      }
    );
  }, []);
  
  return {
    location,
    loading,
    error,
    requestLocation
  };
}
Discovery Hook
typescript// src/hooks/useDiscovery.ts
import { useState, useEffect } from 'react';
import { getFunctions, httpsCallable } from 'firebase/functions';
import { app } from '@/lib/firebase';

const functions = getFunctions(app);
const discoverActivities = httpsCallable(functions, 'discoverActivities');

export function useDiscovery(location: Location | null, filters: any) {
  const [activities, setActivities] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [metadata, setMetadata] = useState<any>(null);
  
  const fetchActivities = useCallback(async () => {
    if (!location) return;
    
    setLoading(true);
    setError(null);
    
    try {
      const result = await discoverActivities({
        lat: location.lat,
        lng: location.lng,
        radius: filters.radius || 10,
        filters: filters
      });
      
      const data = result.data as any;
      
      setActivities(data.results);
      setMetadata(data.metadata);
    } catch (err) {
      setError(err as Error);
    } finally {
      setLoading(false);
    }
  }, [location, filters]);
  
  useEffect(() => {
    fetchActivities();
  }, [fetchActivities]);
  
  return {
    activities,
    loading,
    error,
    metadata,
    refetch: fetchActivities
  };
}
